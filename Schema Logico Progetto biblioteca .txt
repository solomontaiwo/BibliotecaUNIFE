CREAZIONE TABELLE 


CREATE SCHEMA IF NOT EXISTS Biblioteca;
USE Biblioteca;


CREATE TABLE Libro (
        CodLibro        INT NOT NULL AUTO_INCREMENT,
Titolo                VARCHAR(100),
        ISBN                VARCHAR(20),
Lingua                VARCHAR(30),
AnnoPubb        YEAR,
        CodDip        INT,


        PRIMARY KEY (CodLibro),
        FOREIGN KEY (CodDip) REFERENCES Dipartimento(CodDip)
ON DELETE CASCADE 
ON UPDATE CASCADE
);

CREATE TABLE Autore (
        CodAutore                INT NOT NULL AUTO_INCREMENT,
Nome                         VARCHAR(30),
        Cognome                 VARCHAR(30),
        DataNascita                 DATE,
        LuogoNascita                 VARCHAR(30),


        PRIMARY KEY (CodAutore)
);


CREATE TABLE Utente (
        NMatricola         INT NOT NULL,
        Nome                 VARCHAR(30),
        Cognome         VARCHAR(30),
        Via                 VARCHAR(30),
        NCivico        VARCHAR(5),
        Cap                CHAR(5),
        Città                VARCHAR(30),
        NTelefono         VARCHAR(15),
        
        PRIMARY KEY (NMatricola),
        UNIQUE (NTelefono) 
);


CREATE TABLE Dipartimento (
        CodDip        INT NOT NULL AUTO_INCREMENT,
        Nome                 VARCHAR(100),
        Via                 VARCHAR(30),
NCivico        VARCHAR(30),        
        Cap                CHAR(5),
        Città                VARCHAR(30),
        
        PRIMARY KEY (CodDip),
        UNIQUE (Nome)
);


CREATE TABLE Scrivere (
        Codice            INT NOT NULL,
CodLibro        INT,
        CodAutore        INT,
        
        PRIMARY KEY (Codice),
        FOREIGN KEY (CodLibro) REFERENCES Libro (CodLibro)
ON DELETE CASCADE 
ON UPDATE CASCADE,
        FOREIGN KEY (CodAutore) REFERENCES Autore (CodAutore)
        ON DELETE CASCADE 
ON UPDATE CASCADE
);


CREATE TABLE Prestito (
        CodPrestito     INT NOT NULL AUTO_INCREMENT,
Restituzione    BOOLEAN,
        DataUscita        DATE,
        CodLibro        INT,
        NMatricola        INT,
        
        PRIMARY KEY (CodPrestito),
        FOREIGN KEY (CodLibro) REFERENCES Libro (CodLibro)
ON DELETE CASCADE 
ON UPDATE CASCADE,
        FOREIGN KEY (NMatricola) REFERENCES Utente (NMatricola)
ON DELETE CASCADE 
ON UPDATE CASCADE
);
